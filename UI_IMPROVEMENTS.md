const emailData = {

  pizza: orderData?.pizza?.nameEn || 'Pizza',
  size: data.size,
  total: data.totalPrice,
  message: `New order from ${data.customerName}...`

await fetch('https://formspree.io/f/meorndkv', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },

});
